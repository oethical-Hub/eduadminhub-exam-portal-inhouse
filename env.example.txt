# ============================================================
# APPLICATION MODE
# ============================================================
# Options: "standalone" or "integrated"
# - standalone: Uses your own Node.js + MongoDB backend
# - integrated: Uses EduAdminHub backend with OTP authentication
# 
# Note: This is the default mode. Users can also switch modes
# via URL query parameter: /login?mode=standalone or /login?mode=integrated

NEXT_PUBLIC_MODE=standalone

# NEXT_PUBLIC_STANDALONE_API_URL=https://api.your-domain.com/api/v1
